# Obtain the x and y coordinates for each of the three city types
urban_df =merged_data_df.loc[merged_data_df['type']=='Urban'].groupby([merged_data_df["city"]])
suburban_df =merged_data_df.loc[merged_data_df['type']=='Suburban'].groupby([merged_data_df["city"]])
rural_df =merged_data_df.loc[merged_data_df['type']=='Rural'].groupby([merged_data_df["city"]])
#urban_df.head()

urban_x = urban_df["ride_id"].count()
urban_y=urban_df["fare"].mean()
urban_size = urban_df["driver_count"].mean()

suburban_x = suburban_df["ride_id"].count()
suburban_y = suburban_df["fare"].mean()
suburban_size = suburban_df["driver_count"].mean()

rural_x = rural_df["ride_id"].count()
rural_y = rural_df["fare"].mean()
rural_size = rural_df["driver_count"].mean()

# Build the scatter plots for each city types
plt.scatter(urban_x, urban_y, label = "Urban",s=urban_size*10, color=["coral"], edgecolor="black", alpha = 0.75, marker="o")
plt.scatter(suburban_x, suburban_y, label = "Suburban", s=suburban_size*10, color = ["gold"], edgecolor = "black", alpha = 0.75, marker = "o")
plt.scatter(rural_x, rural_y, label = "Rural", s=rural_size*10, color = ["lightskyblue"], edgecolor = "black", alpha = 0.75, marker= 'o')

# Incorporate the other graph properties
plt.grid()
plt.xlabel("Total Number of Rides (per City)")
plt.ylabel("Average Fare ($)")
# Create a legend
legend = plt.legend(fontsize=8, title = "Type of City", loc = "best")
legend.legendHandles[0]._sizes = [30]
legend.legendHandles[1]._sizes = [30]
legend.legendHandles[2]._sizes = [30]

# Incorporate a text label regarding circle size
plt.text(42,30,"Note: Circle size correlates with driver count per city.")

# Save Figure
plt.savefig("PyberRideSharingData2016.png", bbox_inches='tight')
# Show plot
plt.show()

# Calculate Type Percents 
cityTypes_grouped = merged_data_df.groupby(['type'])
fare_sum = cityTypes_grouped['fare'].sum()

# Build Pie Chart
labels_type = ['Rural', 'Suburban','Urban']
explode =(0,0,0.1)
colors = ["lightskyblue", "gold", "lightcoral"]
plt.title("% of Total Fares by City Type")
plt.pie(fare_sum, explode = explode, labels = labels_type, colors = colors, autopct ="%2.2f%%", shadow = True, startangle = 140)

# Save Figure
plt.savefig("TotalFaresbyCityType.png")
plt.show()
